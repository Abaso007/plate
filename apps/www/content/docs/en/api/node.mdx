```ts
/**
 * The `TNode` union type represents all of the different types of nodes that
 * occur in a Slate document tree.
 */
```

### `ancestor`

Get the node at a specific path, asserting that it's an ancestor node.

<APIParameters>
  <APIItem name="root" type="TNode">
    The root node to start from.
  </APIItem>
  <APIItem name="path" type="Path">
    The path to the ancestor node.
  </APIItem>
</APIParameters>

<APIReturns>
  Returns `N | undefined` - The ancestor node if found, `undefined` otherwise.
</APIReturns>

### `ancestors`

```ts
/**
 * Return a generator of all the ancestor nodes above a specific path.
 *
 * By default the order is bottom-up, from lowest to highest ancestor in the
 * tree, but you can pass the `reverse: true` option to go top-down.
 */
```

<APIParameters>
  <APIItem name="root" type="TNode">
    The root node to start from.
  </APIItem>
  <APIItem name="path" type="Path">
    The path to get ancestors for.
  </APIItem>
  <APIItem name="options" type="NodeAncestorsOptions" optional>
    <APISubList>
      <APISubListItem parent="options" name="reverse" type="boolean" optional>
        If true, returns ancestors top-down instead of bottom-up.
      </APISubListItem>
    </APISubList>
  </APIItem>
</APIParameters>

<APIReturns>
  Returns `Generator<NodeEntry<N>, void, undefined>` - A generator of node entries.
</APIReturns>

### `children`

Iterate over the children of a node at a specific path.

<APIParameters>
  <APIItem name="root" type="TNode">
    The root node.
  </APIItem>
  <APIItem name="path" type="Path">
    The path to the parent node.
  </APIItem>
  <APIItem name="options" type="NodeChildrenOptions" optional>
    <APISubList>
      <APISubListItem parent="options" name="reverse" type="boolean" optional>

      </APISubListItem>
    </APISubList>
  </APIItem>
</APIParameters>

<APIReturns>
  Returns `Generator<NodeEntry<N>, void, undefined>` - A generator of node entries.

### `descendants`

Return a generator of all the descendant node entries inside a root node.

<APIParameters>
  <APIItem name="root" type="TNode">
    The root node.
  </APIItem>
  <APIItem name="options" type="NodeDescendantsOptions" optional>
    <APISubList>
      <APISubListItem parent="options" name="from" type="Path" optional>
        Starting path.
      </APISubListItem>
      <APISubListItem parent="options" name="to" type="Path" optional>
        Ending path.
      </APISubListItem>
      <APISubListItem parent="options" name="reverse" type="boolean" optional>
        If true, iterates in reverse order.
      </APISubListItem>
      <APISubListItem parent="options" name="pass" type="function" optional>

      </APISubListItem>
    </APISubList>
  </APIItem>
</APIParameters>

<APIReturns>
  Returns `Generator<NodeEntry<N>, void, undefined>` - A generator of node entries.
</APIReturns>

### `elements`

```ts
/**
 * Return a generator of all the element nodes inside a root node. Each

```

<APIParameters>
  <APIItem name="root" type="TNode">
    The root node.
  </APIItem>
  <APIItem name="options" type="NodeElementsOptions" optional>
    Similar options to `descendants`.
  </APIItem>
</APIParameters>

<APIReturns>
  Returns `Generator<NodeEntry<N>, void, undefined>` - A generator of element entries.
</APIReturns>

## Types

### `Node`

```ts
/**
 * The `Node` union type represents all of the different types of nodes that
 * occur in a Slate document tree.
 */
type Node = TNode;
```

### `Descendant`

```ts
/**
 * The `Descendant` union type represents nodes that are descendants in the
 * tree. It is returned as a convenience in certain cases to narrow a value
 * further than the more generic `Node` union.
 */
type Descendant = TElement | TText;
```

### `Ancestor`

```ts
/**
 * The `Ancestor` union type represents nodes that are ancestors in the tree. It
 * is returned as a convenience in certain cases to narrow a value further than
 * the more generic `Node` union.
 */
type Ancestor = Editor | TElement;
```

### `NodeProps`

```ts
/** Convenience type for returning the props of a node. */
type NodeProps<N extends TNode = TNode> = N extends Editor
  ? Omit<N, 'children'>
  : N extends TElement
    ? Omit<N, 'children'>
    : Omit<N, 'text'>;
```

### `NodeEntry`

```ts
/** A tuple type representing a node and its location in the tree. */
type NodeEntry<T extends Node = Node> = [T, Path];
```

### Options Interfaces

```ts
interface NodeAncestorsOptions {
  /** If true, the paths are returned in reverse order. */
  reverse?: boolean;
}

interface NodeChildrenOptions {
  /** Get children starting from this index (inclusive) */
  from?: number;
  reverse?: boolean;
  /** Get children up to this index (exclusive) */
  to?: number;
}

interface NodeDescendantsOptions<N extends TNode> {
  from?: Path;
  pass?: (entry: NodeEntry<DescendantOf<N>>) => boolean;
  reverse?: boolean;
  to?: Path;
}
```
