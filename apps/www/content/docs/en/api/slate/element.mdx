```ts
/**
 * `TElement` objects are a type of node in a Slate document that contain other
 * element nodes or text nodes. They can be either "blocks" or "inlines"

 */
```

## Element API

### `isElementType`

```ts
/**
 * Check if a value implements the `TElement` interface and has elementKey
 * with selected value. Default it check to `type` key value
 */
```

<APIParameters>
  <APIItem name="value" type="any">
    The value to check.
  </APIItem>
  <APIItem name="elementVal" type="string">
    The value to match against.
  </APIItem>
  <APIItem name="elementKey" type="string" optional>
    The key to check. Defaults to 'type'.
  </APIItem>
</APIParameters>

<APIReturns>
  Returns `boolean` - `true` if the value is an element with matching type.
</APIReturns>

### `matches`

```ts

 * Check if an element matches set of properties.
 *
 * Note: this checks custom properties, and it does not ensure that any
 * children are equivalent.
 */
```

<APIParameters>
  <APIItem name="element" type="TElement">
    The element to check.
  </APIItem>
  <APIItem name="props" type="Partial<TElement>">
    The properties to match against.
  </APIItem>
</APIParameters>

<APIReturns>
  Returns `boolean` - `true` if the element matches all properties.
</APIReturns>

## Types

### `TElement`

```ts
/**
 * `TElement` objects are a type of node in a Slate document that contain other
 * element nodes or text nodes. They can be either "blocks" or "inlines"
 * depending on the Slate editor's configuration.
 */
type TElement = {
  children: Descendant[];
  type: string;
} & UnknownObject;
```

<APIAttributes>
  <APIItem name="TElement" type="object">
    Properties:
    - `children: Descendant[]` - Array of child nodes
    - `type: string` - Element type identifier
    - `[key: string]: unknown` - Additional custom properties
  </APIItem>
</APIAttributes>

### `Element`

```ts
/**
 * `Element` objects are a type of node in a Slate document that contain other
 * element nodes or text nodes. They can be either "blocks" or "inlines"
 * depending on the Slate editor's configuration.
 */
type Element = TElement;
```

### `ElementOrTextOf`

```ts
/** Element or text of an editor. */
type ElementOrTextOf<E extends Editor> = ElementOf<E> | TextOf<E>;
```

<APIAttributes>
  <APIItem name="ElementOrTextOf<E>" type="ElementOf<E> | TextOf<E>">
    Generic type for elements or text nodes from a specific editor type.
  </APIItem>
</APIAttributes>

### `ElementOrTextIn`

```ts
type ElementOrTextIn<V extends Value> = ElementIn<V> | TextIn<V>;
```

<APIAttributes>
  <APIItem name="ElementOrTextIn<V>" type="ElementIn<V> | TextIn<V>">
    Generic type for elements or text nodes from a specific value type.
  </APIItem>
</APIAttributes>

### `ElementOf`

```ts
/** A utility type to get all the element nodes type from a root node. */
type ElementOf<N extends TNode> = Editor extends N
  ? TElement
  : TElement extends N
    ? TElement
    : N extends Editor
      ? ElementOf<N['children'][number]> | Extract<N['children'][number], TElement>
      : N extends TElement
        ? ElementOf<N['children'][number]> | Extract<N['children'][number], TElement> | N
        : never;
```

### `ElementIn`

```ts
type ElementIn<V extends Value> = ElementOf<V[number]>;
```

<APIAttributes>
  <APIItem name="ElementIn<V>" type="ElementOf<V[number]>">
    Generic type that extracts element types from a value type.
  </APIItem>
</APIAttributes>
