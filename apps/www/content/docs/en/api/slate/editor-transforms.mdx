Transforms are helper functions operating on the document.

## Node Operations

### `duplicateNodes`

Duplicate nodes at a location.

<API name="duplicateNodes">
<APIParameters>
  <APIItem name="options" type="DuplicateNodesOptions" optional />
</APIParameters>

<APIOptions type="DuplicateNodesOptions">
  <APIItem name="...options" type="InsertNodesOptions" optional>
    `insertNodes` options.
  </APIItem>
  <APIItem name="at" type="At" optional>
    Location to get nodes from and insert after. Default: selection
  </APIItem>
  <APIItem name="block" type="boolean" optional>
    If true, duplicate blocks above location. Ignored if `nodes` is provided
  </APIItem>
  <APIItem name="nodes" type="NodeEntry[]" optional>
    Specific nodes to duplicate. If provided, ignores `block` option
  </APIItem>
</APIOptions>
</API>

## Mark Operations

### `addMark`

Add a custom property to the leaf text nodes in the current selection.

<API name="addMark">
<APIParameters>
  <APIItem name="key" type="string">
    The key of the mark to add.
  </APIItem>
  <APIItem name="value" type="any">
    The value to set for the mark property.
  </APIItem>
</APIParameters>
</API>

### `addMarks`

Add multiple marks to the leaf text nodes in the current selection.

```ts
editor.tf.addMarks({ bold: true, italic: true })
editor.tf.addMarks({ bold: subscript }, { remove: 'superscript' })
editor.tf.addMarks({ bold: true }, { remove: ['italic', 'underline'] })
```

<API name="addMarks">
<APIParameters>
  <APIItem name="marks" type="Record<string, any>">
    An object mapping mark keys to their values.
  </APIItem>
  <APIItem name="options" type="AddMarksOptions" optional />
</APIParameters>

<APIOptions type="AddMarksOptions">
  <APIItem name="remove" type="string[] | string" optional>
    Mark keys to remove before adding new ones. Useful for mutually exclusive marks.
  </APIItem>
</APIOptions>
</API>

### `removeMark`

Remove a custom property from all of the leaf text nodes within non-void nodes or void nodes that `editor.markableVoid()` allows in the current selection.

<API name="removeMark">
<APIParameters>
  <APIItem name="key" type="string">
    The key of the mark to remove.
  </APIItem>
</APIParameters>
</API>

### `removeMarks`

Remove marks from text nodes.

```ts
// editor.tf.removeMarks() // Remove all marks from editor.marks
// editor.tf.removeMarks('bold') // Remove bold mark at selection
// editor.tf.removeMarks(['bold', 'italic']) // Remove multiple marks at selection
// editor.tf.removeMarks('bold', { at: range }) // Remove bold in range
```

<API name="removeMarks">
<APIParameters>
  <APIItem name="keys" type="string[] | string" optional>
    The specific mark key(s) to remove. If not provided and selection is
    collapsed, removes all marks from `editor.marks`.
  </APIItem>
  <APIItem name="options" type="RemoveMarksOptions" optional />
</APIParameters>

<APIOptions type="RemoveMarksOptions">
  <APIItem name="...options" type="UnsetNodesOptions" optional>
    `unsetNodes` options.
  </APIItem>
  <APIItem name="at" type="TRange" optional>
    Range where the mark(s) will be removed. Defaults to current selection.
  </APIItem>
  <APIItem name="shouldChange" type="boolean" optional>
    When true, trigger onChange if selection is collapsed.
  </APIItem>
</APIOptions>
</API>

### `toggleMark`

Toggle a mark on the leaf text nodes in the current selection.

```ts
// editor.tf.toggleMark('bold') // Toggle bold mark
// editor.tf.toggleMark('subscript', { remove: 'superscript' }) // Add subscript, remove superscript
```

<API name="toggleMark">
<APIParameters>
  <APIItem name="key" type="string">
    The key of the mark to toggle.
  </APIItem>
  <APIItem name="value" type="any" optional>
    The value to set if the mark is being added.
  </APIItem>
  <APIItem name="options" type="ToggleMarkOptions" optional />
</APIParameters>

<APIOptions type="ToggleMarkOptions">
  <APIItem name="remove" type="string[] | string" optional>
    Mark keys to remove when adding the mark.
  </APIItem>
</APIOptions>
</API>

### `select`

Set the selection to a new value. When a selection already exists, this method will update the existing value.

```ts
// editor.tf.select(at) // Select at location
// editor.tf.select(at, { edge: 'end' }) // Select end of block above
// editor.tf.select(at, { edge: 'start' }) // Select start of block above
```

<API name="select">
<APIParameters>
  <APIItem name="at" type="At">
    The location to select.
  </APIItem>
  <APIItem name="options" type="SelectOptions" optional />
</APIParameters>

<APIOptions type="SelectOptions">
  <APIItem name="edge" type="'start' | 'end'" optional>
    Select edge of the block above location.
  </APIItem>
  <APIItem name="focus" type="boolean" optional>
    If true, focus the editor before selecting.
  </APIItem>
  <APIItem name="next" type="boolean" optional>
    Select start of next sibling.
  </APIItem>
  <APIItem name="previous" type="boolean" optional>
    Select end of previous sibling.
  </APIItem>
</APIOptions>
</API>
