The Editor API provides a set of helper functions for querying and manipulating the editor state.

## `QueryOptions`

Common options used across multiple query methods:

<APIParameters>
  <APIItem name="at" type="At" optional>
    Location to start searching from. Defaults to current selection.
  </APIItem>
  <APIItem name="id" type="boolean | string" optional>
    Match the node by id. `true` will match all nodes with an id.
  </APIItem>
  <APIItem name="block" type="boolean" optional>
    Match block nodes.
  </APIItem>
  <APIItem name="empty" type="boolean" optional>
    When true, match only empty nodes. When false, match only non-empty nodes.
  </APIItem>
  <APIItem name="match" type="Predicate<NodeIn<V>>" optional>
    Match the node.
  </APIItem>
  <APIItem name="text" type="boolean" optional>
    When true, match only text nodes.
  </APIItem>
</APIParameters>

## `QueryMode`

<APIParameters>
  <APIItem name="mode" type="'all' | 'highest' | 'lowest'" optional>
    - 'all' (default): Return all matching nodes
    - 'highest': In a hierarchy of nodes, only return the highest level matching nodes
    - 'lowest': In a hierarchy of nodes, only return the lowest level matching nodes
  </APIItem>
</APIParameters>

### `above`

Get the matching ancestor above a location in the document.

<APIParameters>
  <APIItem name="options" type="EditorAboveOptions<V>" optional>
    <APISubList>
      <APISubListItem parent="options" name="...options" type="QueryOptions<V>">
        Common query options.
      </APISubListItem>
      <APISubListItem parent="options" name="mode" type="QueryMode">
        Query mode options.
      </APISubListItem>
      <APISubListItem parent="options" name="voids" type="boolean" optional>
        Whether to include void nodes in the search
      </APISubListItem>
    </APISubList>
  </APIItem>
</APIParameters>

<APIReturns>
  Returns a `NodeEntry<N extends AncestorIn<V>>` tuple containing the matching ancestor node and its path, or `undefined` if no match is found.
</APIReturns>

### `block`

Get the block at a location or find the first block that matches options.

```ts
editor.api.block() // Get block above selection
editor.api.block({ above: true }) // Get block above selection
editor.api.block({ at: [0, 0] }) // Get block at [0, 0]
editor.api.block({ at: [0, 0], above: true }) // Get block at [0]
editor.api.block({ highest: true }) // Get highest block at selection
```

<APIParameters>
  <APIItem name="options" type="EditorBlockOptions<V>" optional>
    <APISubList>
      <APISubListItem parent="options" name="...options" type="QueryOptions<V>">
        Common query options.
      </APISubListItem>
      <APISubListItem parent="options" name="at" type="At | Span" optional>
        The location to query at. Defaults to current selection.
      </APISubListItem>
      <APISubListItem parent="options" name="ignoreNonSelectable" type="boolean" optional>
        Whether to ignore non-selectable nodes during traversal.
      </APISubListItem>
      <APISubListItem parent="options" name="reverse" type="boolean" optional>
        Whether to traverse in reverse order.
      </APISubListItem>
      <APISubListItem parent="options" name="universal" type="boolean" optional>
        Whether to ensure the operation works universally across all nodes.
      </APISubListItem>
      <APISubListItem parent="options" name="above" type="boolean" optional>
        If true, get the block above the location.
      </APISubListItem>
      <APISubListItem parent="options" name="highest" type="boolean" optional>
        If true, get the highest block at the location.
      </APISubListItem>
    </APISubList>
  </APIItem>
</APIParameters>

<APIReturns>
  Returns the matching block node entry or `undefined` if no match is found.
</APIReturns>

### `isEmpty`

Check if an element is empty, accounting for void nodes.

```ts
editor.api.isEmpty() // Check if editor is empty
editor.api.isEmpty(at) // Check if nodes at location are empty
editor.api.isEmpty(at, { after: true }) // Check if text after location is empty
editor.api.isEmpty(at, { block: true }) // Check if block above location is empty
```

<APIParameters>
  <APIItem name="options" type="EditorEmptyOptions" optional>
    <APISubList>

### `isAt`

Check if a location (point/range) is at a specific position.

```ts
// For ranges:
editor.api.isAt({ text: true }) // Check if range is in single text node
editor.api.isAt({ block: true }) // Check if range is in single block
editor.api.isAt({ blocks: true }) // Check if range is across multiple blocks
editor.api.isAt({ start: true }) // Check if range starts at block start
editor.api.isAt({ end: true }) // Check if range ends at block end

// For points:
editor.api.isAt({ word: true }) // Check relative to word boundaries
editor.api.isAt({ start: true }) // Check if at start
editor.api.isAt({ end: true }) // Check if at end
```

<APIParameters>
  <APIItem name="options" type="object">
    <APISubList>
      <APISubListItem parent="options" name="at" type="At" optional>
        The location to check. Defaults to current selection.
      </APISubListItem>
      <APISubListItem parent="options" name="text" type="boolean" optional>
        Check if range is in single text node.
      </APISubListItem>
      <APISubListItem parent="options" name="block" type="boolean" optional>
        Check if range is in single block.
      </APISubListItem>
      <APISubListItem parent="options" name="blocks" type="boolean" optional>
        Check if range is across multiple blocks.
      </APISubListItem>
      <APISubListItem parent="options" name="start" type="boolean" optional>
        Check if at start position.
      </APISubListItem>
      <APISubListItem parent="options" name="end" type="boolean" optional>
        Check if at end position.
      </APISubListItem>
      <APISubListItem parent="options" name="word" type="boolean" optional>
        Check relative to word boundaries.
      </APISubListItem>
    </APISubList>
  </APIItem>
</APIParameters>

<APIReturns>
  Returns `true` if the location matches the specified position criteria.
</APIReturns>

Get a property value from a list of nodes. Returns undefined if the property value is not consistent across all nodes.

<APIParameters>
  <APIItem name="options" type="EditorPropOptions<V>">

      <APISubListItem parent="options" name="nodes" type="TElement[]">

      <APISubListItem parent="options" name="defaultValue" type="string" optional>

### `unhangRange`

Convert a range into a non-hanging one.

```ts
// A "hanging" range is one created by the browser's "triple-click" selection behavior.
// When triple-clicking a block, the browser selects from the start of that block to
// the start of the next block. The range thus "hangs over" into the next block.
```

<APIParameters>
  <APIItem name="range" type="Range">
    The range to unhang.
  </APIItem>
  <APIItem name="options" type="EditorUnhangRangeOptions" optional>
    <APISubList>

      </APISubListItem>
    </APISubList>
  </APIItem>
</APIParameters>

<APIReturns>
  Returns `Range` - A new range with the end point moved backwards if it was hanging.
</APIReturns>
