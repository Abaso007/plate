## Features

- Insert horizontal lines to separate content or indicate topic shifts
- Type three dashes (`---`) at a new line start to transform into a horizontal rule

</PackageInfo>

## Installation

```bash
npm install @udecode/plate-horizontal-rule
```

## Usage

```tsx
import { insertNodes, setNodes, KEYS } from '@udecode/plate';
import { AutoformatPlugin } from '@udecode/plate-autoformat/react';
import { ParagraphPlugin } from '@udecode/plate/react';
import { HorizontalRulePlugin } from '@udecode/plate-horizontal-rule/react';
import { SelectOnBackspacePlugin } from '@udecode/plate/react';

const plugins = [
  // ...otherPlugins,
  HorizontalRulePlugin,
  SelectOnBackspacePlugin.configure({
    options: { query: { allow: [KEYS.hr] } },
  }),
  AutoformatPlugin.configure({
    options: {
      rules: [
        {
          mode: 'block',
          type: KEYS.hr,
          match: ['---', 'â€”-', '___ '],
          format: (editor) => {
            setNodes(editor, { type: KEYS.hr });
            insertNodes(editor, {
              type: KEYS.p,
              children: [{ text: '' }],
            });
          },
        },
      ],
    },
  }),
];
```

## Plugins

### `HorizontalRulePlugin`

Plugin for inserting horizontal rules to separate content.
