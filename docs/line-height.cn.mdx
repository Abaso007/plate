## 功能

- 为选中的块设置行高。

</PackageInfo>

## 安装

```bash
npm install @udecode/plate-basic-styles
```

## 使用

```tsx
import { LineHeightPlugin } from '@udecode/plate-basic-styles/react';

const plugins = [
  // ...其他插件,
  LineHeightPlugin,
];
```

## 插件

### `LineHeightPlugin`

用于在块上设置行高的插件。

## API

### `setLineHeight`

为编辑器中的选中节点设置行高。

<API name="setLineHeight">
<APIOptions type="SetLineHeightOptions">
  <APIItem name="value" type="number">
    要设置的行高值。
  </APIItem>
  <APIItem name="setNodesOptions" type="SetNodesOptions">
    设置节点的额外选项。
  </APIItem>
</APIOptions>
</API>
